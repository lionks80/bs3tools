Tutorial 1

basic table.

Create Grid & Data Bind

// tutorial_1_table
$('#tutorial1_table').bs3table({
	columns : [ {
		header : 'ID',
		headerAlign : 'center',
		dataIndex : 'id',
		dataAlign : 'right',
		width : 40
	}, {
		header : 'Name',
		dataIndex : 'name',
		headerAlign : 'center',
		dataAlign : 'center',
	}, {
		header : 'Age',
		dataIndex : 'age',
		headerAlign : 'center',
		dataAlign : 'center'
	}, {
		header : 'Adult',
		headerAlign : 'center',
		dataAlign : 'center',
		renderer : function(rowData, rowIndex, td, tr) {
			if (rowData.age >= 20) {
				tr.addClass("success");
				return "YES";
			} else {
				tr.addClass("danger");
				return "NO";
			}
		}
	} ],
	listener : {
		rowClick : function(rowIndex, rowData, tr) {
			alert('rowIndex: ' + rowIndex + ', name: ' + rowData.name);
		}
	},
	tableCss : [ "table", "table-bordered"],
	selectbox : {
		useSelect : false,
		multiSelect : false,
		selectByRowClick : false,
		selectedRowCss : "active"
	}
});

// Data Bind
$('#tutorial1_table').bs3table('dataBind', people);

$('#btn_tutorial1_dataBind').click(function() {
	$('#tutorial1_table').bs3table('dataBind', people);
});
$('#btn_tutorial1_dataBind_add').click(function() {
	$('#tutorial1_table').bs3table('dataBind', people, true);
});
$('#btn_tutorial1_clear').click(function() {
	$('#tutorial1_table').bs3table('clear');
});

Tutorial 1 Example

check point

  • align: text-align of table(header, data)
  • rowClick: check row click Event
  • renderer: show data by user function. check tr rendering.
  • databind: data clear and bind
  • dataBind(add): add data to table
  • clear: clear table body

Tutorial 2

use single select table.

Create Grid & Data Bind

// tutorial2_table
$('#tutorial2_table').bs3table({
	columns : [ {
		header : 'ID',
		headerAlign : 'center',
		sort: true,
		sortName:'id',
		dataIndex : 'id',
		dataAlign : 'right',
		width : 60
	}, {
		header : 'Name',
		dataIndex : 'name',
		headerAlign : 'center',
		dataAlign : 'center'
	}, {
		header : 'Age',
		dataIndex : 'age',
		headerAlign : 'center',
		dataAlign : 'center'
	}, {
		header : 'Adult',
		headerAlign : 'center',
		dataAlign : 'center',
		sort : true,
		sortName : 'age',
		renderer : function(rowData, rowIndex, td, tr) {
			if (rowData.age >= 20) {
				td.addClass("success");
				return "YES";
			} else {
				td.addClass("danger");
				return "NO";
			}
		}
	} ],
	listener : {
		sortChange : function(sortName, sortASC) {
			var sortFn = function(a, b){return a[sortName] - b[sortName];}; 
			if (sortASC == false) {
				sortFn = function(a, b){return b[sortName] - a[sortName];};
			}
			
			people.sort(sortFn);
			$('#tutorial2_table').bs3table('dataBind', people);
			
		}
	},
	tableCss : [ "table", "table-bordered", "table-hover" ],
	selectbox : {
		useSelect : true,
		multiSelect : false,
		selectByRowClick : true,
		selectedRowCss : "active"
	}
});

// Data Bind
$('#tutorial2_table').bs3table('setSortOption', {
	"sortName" : "id",
	"sortASC" : true
});
$('#tutorial2_table').bs3table('dataBind', people);

$('#btn_tutorial2_getSortOption').click(function() {
	sortOption = $('#tutorial2_table').bs3table('getSortOption');
	alert("sortName: " + sortOption[0].sortName + " / "  + "sortASC: " + sortOption[0].sortASC);
});

$('#btn_tutorial2_getSelectedData').click(function() {
	selectedData = $('#tutorial2_table').bs3table('getSelectedData');
	
	if (selectedData.length == 0) {
		alert('not selected');
	} else {
		alert(selectedData[0].name);			
	}
});

Tutorial 2 Example

check point

  • setSortOption: set sortOption to table. not fired sortChange event
  • check sortChange: click sortable header.
  • getSortOption: get current sortOptions
  • renderer: check td rendering
  • single select: only one row can be seleted.
  • getSelectedData: only one row can be seleted.

Usage

Install


		
<link type="text/css" rel="stylesheet" href="bs3table-sort.css"/>
<script src="bs3tools.js"></script>

Create Grid

$('target').bs3table(options)

default options

{
	var defaultSetting = {
		columns : [{
			header : 'ID',
			headerAlign : 'center',
			headerCss : ['cssClassName'] 
			sort: true,
			sortName:'id',
			dataIndex : 'id',
			dataAlign : 'right',
			width : 60
		},{
			header : 'Adult',
			headerAlign : 'center',
			dataAlign : 'center',
			sort : true,
			sortName : 'age',
			renderer : function(rowData, rowIndex, td, tr) {
				if (rowData.age >= 20) {
					td.addClass("success");
					return "YES";
				} else {
					td.addClass("danger");
					return "NO";
				}
			}
		}],
		tableCss : [ "table" ],
		listener : {
			rowClick : function(rowIndex, rowData, tr) {
			},
			sortChange : function(sortName, sortASC) {
			}
		},
		selectbox : {
			useSelect : false,
			multiSelect : false,
			selectByRowClick : false,
			selectedRowCss : "active"
		}
	};
}

options

Name Type Description
columns column Array 각 컬럼들을 정의한다.
tableCss string array table에 적용될 css Class 들을 정의.
selectbox string selectbox 사용 시 옵션들.
listener json object event listener 클래스

options.columns

Name Type Description
header string 헤더명.
headerAlign string 헤더 정렬 ['left', 'center', 'right'].
sort boolean 현재 컬럼에서 정렬 사용 여부.
sortName string 정렬 사용시 사용할 프로퍼티 이름(생략시 dataIndex 사용)
dataIndex string 바인딩 된 데이터에서 현재 컬럼에 사용될 프로퍼티 이름.
dataAlign string 데이터 정렬 ['left', 'center', 'right'].
width mixed 컬럼 넓이
renderer function(record, rowIndex) 해당 셀에 append 시킬 문자 및 객체를 반환하도록 작성한다.

options.listener

Name Type Description
rowClick function(rowIndex, rowData, tr) 로우 클릭 시 동작함수
sortChange function(sortName, sortASC) header의 정렬을 클릭 시 발생됨.

Methods

Return Method Description
void $(target).bs3table('dataBind', data, addRow) json object array data를 그리드에 바인딩. addRow = 추가할지 지우고 새로 바인딩할지 여부
void $(target).bs3table('clear') 현재 그리드의 데이터를 모두 초기화.
record array $(target).bs3table('getSelectedData') 현재 선택된 로우들의 데이터를 반환.
record array $(target).bs3table('setSortOption', { sortName: "id", sortASC : true ) 정렬 이벤트를 발생시키지 않고 초기세팅할때 등에서 사용됨. 화면 상태를 설정한대로 변경하고 현재 솔팅정보를 플러그인에 저장
record array $(target).bs3table('ㅎetSortOption' 현재 테이블에 반영된 솔팅정보(헤더 이벤트 클릭 시 정보를 얻어 솔팅하는 등에서 사용).
